95 template <class K, class MkIndex = MkIndexDefault<K> >
96 class IntSet {
97  IntMap<K, char, MkIndex> in_set;
98  vec<K> xs;
99 
100  public:
101  // Size operations:
102  int size(void) const {
103  return xs.size();
104  }
105  void clear(bool free = false) {
106  if (free)
107  in_set.clear(true);
108  else
109  for (int i = 0; i < xs.size(); i++)
110  in_set[xs[i]] = 0;
111  xs.clear(free);
112  }
113 
114  // Allow inspecting the internal vector:
115  const vec<K>& toVec() const {
116  return xs;
117  }
118 
119  // Vector interface:
120  K operator[](int index) const {
121  return xs[index];
122  }
123 
124  operator const vec<K> &(void) const {
125  return xs;
126  }
127 
128  void insert(K k) {
129  in_set.reserve(k, 0);
130  if (!in_set[k]) {
131  in_set[k] = 1;
132  xs.push(k);
133  }

59  template <typename T>
60  class MyQueue {
61  int max_sz, q_sz;
62  int ptr;
63  int64_t sum;
64  vec<T> q;
65 
66  public:
67  MyQueue(int sz) : max_sz(sz), q_sz(0), ptr(0), sum(0) {
68  assert(sz > 0);
69  q.growTo(sz);
70  }
71  MyQueue(const MyQueue& other)
72  : max_sz(other.max_sz), q_sz(other.q_sz), ptr(other.ptr), sum(other.sum) {
73  other.q.copyTo(q);
74  }
75  inline bool full() const {
76  return q_sz == max_sz;
77  }
78 #ifdef INT_QUEUE_AVG
79  inline T avg() const {
80  assert(full());
81  return sum / max_sz;
82  }
83 #else
84  inline double avg() const {
85  assert(full());
86  return sum / (double) max_sz;
87  }
88 #endif
89  inline void clear() {
90  sum = 0;
91  q_sz = 0;
92  ptr = 0;
93  }
94  void push(T e) {
95  if (q_sz < max_sz)
96  q_sz++;
97  else
98  sum -= q[ptr];
99  sum += e;
100  q[ptr++] = e;
101  if (ptr == max_sz)
102  ptr = 0;
103  }
104  };

422  template <class V>
423  int computeLBD(const V& c) {
424  int lbd = 0;
425 
426  counter++;
427  for (int i = 0; i < c.size(); i++) {
428  int l = level(var(c[i]));
429  if (l != 0 && seen2[l] != counter) {
430  seen2[l] = counter;
431  lbd++;
432  }
433  }
434 
435  return lbd;
436  }

443  static inline void byteDRUP(Lit l) {
444  unsigned int u = 2 * (var(l) + 1) + sign(l);
445  do {
446  *buf_ptr++ = u & 0x7f | 0x80;
447  buf_len++;
448  u = u >> 7;
449  } while (u);
450  *(buf_ptr - 1) &= 0x7f; // End marker of this unsigned number.
451  }
452 
453  template <class V>
454  static inline void binDRUP(unsigned char op, const V& c, FILE* drup_file) {
455  assert(op == 'a' || op == 'd');
456  *buf_ptr++ = op;
457  buf_len++;
458  for (int i = 0; i < c.size(); i++)
459  byteDRUP(c[i]);
460  *buf_ptr++ = 0;
461  buf_len++;
462  if (buf_len > 1048576)
463  binDRUP_flush(drup_file);
464  }
465 
466  static inline void binDRUP_strengthen(const Clause& c, Lit l, FILE* drup_file) {
467  *buf_ptr++ = 'a';
468  buf_len++;
469  for (int i = 0; i < c.size(); i++)
470  if (c[i] != l)
471  byteDRUP(c[i]);
472  *buf_ptr++ = 0;
473  buf_len++;
474  if (buf_len > 1048576)
475  binDRUP_flush(drup_file);
476  }

478  static inline void binDRUP_flush(FILE* drup_file) {
479  fwrite_unlocked(drup_buf, sizeof(unsigned char), buf_len, drup_file);
480  buf_ptr = drup_buf;
481  buf_len = 0;
482  }

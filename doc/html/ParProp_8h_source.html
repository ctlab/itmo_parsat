60 uint64_t ParProp::_prop_search(
61  search::USearch search_p, Prop::prop_callback_t const& callback, Args const&... args) noexcept {
62  /* We apply sequential propagation if the number of samples is less than 8192 */
63  static constexpr size_t SAMPLES_PARALLEL_MIN = 1ULL << 13;
64  static thread_local std::vector<std::future<uint64_t>> futures;
65  if (search_p->size() <= SAMPLES_PARALLEL_MIN) {
66  return _prop_worker_pool
67  .submit([&args..., &search_p, &callback](auto& prop) {
68  return sequential_propagate(prop, args..., std::move(search_p), callback);
69  })
70  .get();
71  }
72  uint32_t num_threads = std::min(search_p->size() / SAMPLES_PARALLEL_MIN, _prop_worker_pool.max_threads());
73  futures.clear();
74  futures.reserve(num_threads);
75  for (uint32_t index = 0; index < num_threads; ++index) {
76  futures.push_back(_prop_worker_pool.submit([&args..., index, &num_threads, &search_p, &callback](auto& prop) {
77  return sequential_propagate(prop, args..., search_p->split_search(num_threads, index), callback);
78  }));
79  }
80  return _sum(futures);
81 }

36 template <class B, class Solver>
37 static void readClause(B& in, Solver& S, vec<Lit>& lits) {
38  int parsed_lit, var;
39  lits.clear();
40  for (;;) {
41  parsed_lit = parseInt(in);
42  if (parsed_lit == 0)
43  break;
44  var = abs(parsed_lit) - 1;
45  while (var >= S.nVars())
46  S.newVar();
47  lits.push((parsed_lit > 0) ? mkLit(var) : ~mkLit(var));
48  }
49 }

51 template <class B, class Solver>
52 static void parse_DIMACS_main(B& in, Solver& S) {
53  vec<Lit> lits;
54  int vars = 0;
55  int clauses = 0;
56  int cnt = 0;
57  for (;;) {
58  skipWhitespace(in);
59  if (*in == EOF)
60  break;
61  else if (*in == 'p') {
62  if (eagerMatch(in, "p cnf")) {
63  vars = parseInt(in);
64  clauses = parseInt(in);
65  // SATRACE'06 hack
66  // if (clauses > 4000000)
67  // S.eliminate(true);
68  } else {
69  printf("PARSE ERROR! Unexpected char: %c\n", *in), exit(3);
70  }
71  } else if (*in == 'c' || *in == 'p')
72  skipLine(in);
73  else {
74  cnt++;
75  readClause(in, S, lits);
76  S.addClause_(lits);
77  }
78  }
79  if (vars != S.nVars())
80  fprintf(stderr, "WARNING! DIMACS header mismatch: wrong number of variables.\n");
81  if (cnt != clauses)
82  fprintf(stderr, "WARNING! DIMACS header mismatch: wrong number of clauses.\n");
83 }

87 template <class Solver>
88 static void parse_DIMACS(gzFile input_stream, Solver& S) {
89  StreamBuffer in(input_stream);
90  parse_DIMACS_main(in, S);

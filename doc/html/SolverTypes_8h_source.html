46 // Lifted booleans:
47 //
48 // NOTE: this implementation is optimized for the case when comparisons between values are mostly

58 class lbool {
59  uint8_t value;
60 
61  public:
62  explicit lbool(uint8_t v) : value(v) {}
63 
64  lbool() : value(0) {}
65  explicit lbool(bool x) : value(!x) {}
66 
67  bool operator==(lbool b) const {
68  return ((b.value & 2) & (value & 2)) | (!(b.value & 2) & (value == b.value));
69  }
70  bool operator!=(lbool b) const {
71  return !(*this == b);
72  }
73  lbool operator^(bool b) const {
74  return lbool((uint8_t)(value ^ (uint8_t) b));
75  }
76 
77  lbool operator&&(lbool b) const {
78  uint8_t sel = (this->value << 1) | (b.value << 3);
79  uint8_t v = (0xF7F755F4 >> sel) & 3;
80  return lbool(v);
81  }
82 
83  lbool operator||(lbool b) const {
84  uint8_t sel = (this->value << 1) | (b.value << 3);
85  uint8_t v = (0xFCFCF400 >> sel) & 3;
86  return lbool(v);
87  }
88 
89  friend int toInt(lbool l);
90  friend lbool toLbool(int v);
91 };

110 class Clause {
111  struct {
112  unsigned mark : 2;
113  unsigned learnt : 1;
114  unsigned has_extra : 1;
115  unsigned reloced : 1;
116  unsigned lbd : 26;
117  unsigned removable : 1;
118  unsigned size : 32;
119  } header;
120  union {
121  Lit lit;
122  float act;
123  uint32_t abs;
124  uint32_t touched;
125  CRef rel;
126  } data[0];
127 
128  friend class ClauseAllocator;
129 
130  // NOTE: This constructor cannot be used directly (doesn't allocate enough memory).
131  template <class V>
132  Clause(const V& ps, bool use_extra, bool learnt) {
133  header.mark = 0;
134  header.learnt = learnt;
135  header.has_extra = learnt | use_extra;
136  header.reloced = 0;
137  header.size = ps.size();
138  header.lbd = 0;
139  header.removable = 1;
140 
141  for (int i = 0; i < ps.size(); i++)
142  data[i].lit = ps[i];
143 
144  if (header.has_extra) {
145  if (header.learnt) {
146  data[header.size].act = 0;
147  data[header.size + 1].touched = 0;
148  } else
149  calcAbstraction();
150  }
151  }
152 
153  public:
154  void calcAbstraction() {
155  assert(header.has_extra);
156  uint32_t abstraction = 0;
157  for (int i = 0; i < size(); i++)
158  abstraction |= 1 << (var(data[i].lit) & 31);
159  data[header.size].abs = abstraction;
160  }
161 
162  int size() const {
163  return header.size;
164  }
165  void shrink(int i) {
166  assert(i <= size());
167  if (header.has_extra)
168  data[header.size - i] = data[header.size];
169  header.size -= i;
170  }
171  void pop() {
172  shrink(1);
173  }
174  bool learnt() const {
175  return header.learnt;
176  }
177  bool has_extra() const {
178  return header.has_extra;
179  }
180  uint32_t mark() const {
181  return header.mark;
182  }
183  void mark(uint32_t m) {
184  header.mark = m;
185  }
186  const Lit& last() const {
187  return data[header.size - 1].lit;
188  }
189 
190  bool reloced() const {
191  return header.reloced;
192  }
193  CRef relocation() const {
194  return data[0].rel;
195  }
196  void relocate(CRef c) {
197  header.reloced = 1;
198  data[0].rel = c;
199  }
200 
201  int lbd() const {
202  return header.lbd;
203  }
204  void set_lbd(int lbd) {
205  header.lbd = lbd;
206  }
207  bool removable() const {
208  return header.removable;
209  }
210  void removable(bool b) {
211  header.removable = b;
212  }
213 
214  // NOTE: somewhat unsafe to change the clause in-place! Must manually call 'calcAbstraction'
215  // afterwards for
216  // subsumption operations to behave correctly.
217  Lit& operator[](int i) {
218  return data[i].lit;
219  }
220  Lit operator[](int i) const {
221  return data[i].lit;
222  }
223  operator const Lit*(void) const {
224  return (Lit*) data;
225  }
226 
227  uint32_t& touched() {
228  assert(header.has_extra && header.learnt);
229  return data[header.size + 1].touched;
230  }
231  float& activity() {
232  assert(header.has_extra);
233  return data[header.size].act;
234  }
235  uint32_t abstraction() const {
236  assert(header.has_extra);
237  return data[header.size].abs;
238  }

240  Lit subsumes(const Clause& other) const;
241  void strengthen(Lit p);

265  template <class Lits>
266  CRef alloc(const Lits& ps, bool learnt = false) {
267  assert(sizeof(Lit) == sizeof(uint32_t));
268  assert(sizeof(float) == sizeof(uint32_t));
269  int extras = learnt ? 2 : (int) extra_clause_field;
270 
271  CRef cid = RegionAllocator<uint32_t>::alloc(clauseWord32Size(ps.size(), extras));
272  new (lea(cid)) Clause(ps, extra_clause_field, learnt);
273

275  }

300  void reloc(CRef& cr, ClauseAllocator& to) {
301  Clause& c = operator[](cr);
302 
303  if (c.reloced()) {
304  cr = c.relocation();
305  return;
306  }
307 
308  cr = to.alloc(c, c.learnt());
309  c.relocate(cr);
310 
311  // Copy extra data-fields:
312  // (This could be cleaned-up. Generalize Clause-constructor to be applicable here instead?)
313  to[cr].mark(c.mark());
314  if (to[cr].learnt()) {
315  to[cr].touched() = c.touched();
316  to[cr].activity() = c.activity();
317  to[cr].set_lbd(c.lbd());
318  to[cr].removable(c.removable());
319  } else if (to[cr].has_extra())

377 template <class Idx, class Vec, class Deleted>
378 void OccLists<Idx, Vec, Deleted>::cleanAll() {
379  for (int i = 0; i < dirties.size(); i++)
380  // Dirties may contain duplicates so check here if a variable is already cleaned:
381  if (dirty[toInt(dirties[i])])
382  clean(dirties[i]);
383  dirties.clear();
384 }

386 template <class Idx, class Vec, class Deleted>
387 void OccLists<Idx, Vec, Deleted>::clean(const Idx& idx) {
388  Vec& vec = occs[toInt(idx)];
389  int i, j;
390  for (i = j = 0; i < vec.size(); i++)
391  if (!deleted(vec[i]))
392  vec[j++] = vec[i];
393  vec.shrink(i - j);
394  dirty[toInt(idx)] = 0;
395 }

461 /*_________________________________________________________________________________________________
462 |
463 | subsumes : (other : const Clause&) -> Lit
464 |
465 | Description:
466 | Checks if clause subsumes 'other', and at the same time, if it can be used to simplify
467 'other' | by subsumption resolution.
468 |
469 | Result:
470 | lit_Error - No subsumption or simplification
471 | lit_Undef - Clause subsumes 'other'
472 | p - The literal p can be deleted from 'other'
473 |________________________________________________________________________________________________@*/
474 inline Lit Clause::subsumes(const Clause& other) const {
475  // if (other.size() < size() || (extra.abst & ~other.extra.abst) != 0)
476  // if (other.size() < size() || (!learnt() && !other.learnt() && (extra.abst & ~other.extra.abst)
477  // != 0))
478  assert(!header.learnt);
479  assert(!other.header.learnt);
480  assert(header.has_extra);
481  assert(other.header.has_extra);
482  if (other.header.size < header.size ||
483  (data[header.size].abs & ~other.data[other.header.size].abs) != 0)
484  return lit_Error;
485 
486  Lit ret = lit_Undef;
487  const Lit* c = (const Lit*) (*this);
488  const Lit* d = (const Lit*) other;
489 
490  for (unsigned i = 0; i < header.size; i++) {
491  // search for c[i] or ~c[i]
492  for (unsigned j = 0; j < other.header.size; j++)
493  if (c[i] == d[j])
494  goto ok;
495  else if (ret == lit_Undef && c[i] == ~d[j]) {
496  ret = c[i];
497  goto ok;
498  }
499 
500  // did not find it
501  return lit_Error;
502  ok:;
503  }
504 
505  return ret;
506 }

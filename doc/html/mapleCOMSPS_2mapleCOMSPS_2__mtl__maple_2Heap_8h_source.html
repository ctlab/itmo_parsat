29 // A heap implementation with support for decrease/increase key.
30 
31 template <class Comp>
32 class Heap {
33  Comp lt; // The heap is a minimum-heap with respect to this comparator
34  vec<int> heap; // Heap of integers
35  vec<int> indices; // Each integers position (index) in the Heap
36 
37  // Index "traversal" functions
38  static inline int left(int i) {
39  return i * 2 + 1;
40  }
41  static inline int right(int i) {
42  return (i + 1) * 2;
43  }
44  static inline int parent(int i) {
45  return (i - 1) >> 1;
46  }
47 
48  void percolateUp(int i) {
49  int x = heap[i];
50  int p = parent(i);
51 
52  while (i != 0 && lt(x, heap[p])) {
53  heap[i] = heap[p];
54  indices[heap[p]] = i;
55  i = p;
56  p = parent(p);
57  }
58  heap[i] = x;
59  indices[x] = i;
60  }
61 
62  void percolateDown(int i) {
63  int x = heap[i];
64  while (left(i) < heap.size()) {
65  int child = right(i) < heap.size() && lt(heap[right(i)], heap[left(i)]) ? right(i) : left(i);
66  if (!lt(heap[child], x))
67  break;
68  heap[i] = heap[child];
69  indices[heap[i]] = i;
70  i = child;
71  }
72  heap[i] = x;
73  indices[x] = i;
74  }
75 
76  public:
77  Heap(const Comp& c) : lt(c) {}
78 
79  int size() const {
80  return heap.size();
81  }
82  bool empty() const {
83  return heap.size() == 0;
84  }
85  bool inHeap(int n) const {
86  return n < indices.size() && indices[n] >= 0;
87  }
88  int operator[](int index) const {
89  assert(index < heap.size());
90  return heap[index];
91  }
92 
93  void decrease(int n) {
94  assert(inHeap(n));
95  percolateUp(indices[n]);
96  }
97  void increase(int n) {
98  assert(inHeap(n));
99  percolateDown(indices[n]);
100  }
101 
102  void copyTo(Heap& copy) const {
103  heap.copyTo(copy.heap);
104  indices.copyTo(copy.indices);
105  }
106 
107  // Safe variant of insert/decrease/increase:
108  void update(int n) {
109  if (!inHeap(n))
110  insert(n);
111  else {
112  percolateUp(indices[n]);
113  percolateDown(indices[n]);
114  }
115  }
116 
117  void insert(int n) {
118  indices.growTo(n + 1, -1);
119  assert(!inHeap(n));
120 
121  indices[n] = heap.size();
122  heap.push(n);
123  percolateUp(indices[n]);
124  }
125 
126  int removeMin() {
127  int x = heap[0];
128  heap[0] = heap.last();
129  indices[heap[0]] = 0;
130  indices[x] = -1;
131  heap.pop();
132  if (heap.size() > 1)
133  percolateDown(0);
134  return x;
135  }
136 
137  // Rebuild the heap from scratch, using the elements in 'ns':
138  void build(const vec<int>& ns) {
139  for (int i = 0; i < heap.size(); i++)
140  indices[heap[i]] = -1;
141  heap.clear();
142 
143  for (int i = 0; i < ns.size(); i++) {
144  indices[ns[i]] = i;
145  heap.push(ns[i]);
146  }
147 
148  for (int i = heap.size() / 2 - 1; i >= 0; i--)
149  percolateDown(i);
150  }
151 
152  void clear(bool dealloc = false) {
153  for (int i = 0; i < heap.size(); i++)
154  indices[heap[i]] = -1;
155  heap.clear(dealloc);
156  }
157 };

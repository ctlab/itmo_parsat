322 static inline word m4ri_swap_bits(word v) {
323  v = ((v >> 1) & 0x5555555555555555ULL) | ((v & 0x5555555555555555ULL) << 1);
324  v = ((v >> 2) & 0x3333333333333333ULL) | ((v & 0x3333333333333333ULL) << 2);
325  v = ((v >> 4) & 0x0F0F0F0F0F0F0F0FULL) | ((v & 0x0F0F0F0F0F0F0F0FULL) << 4);
326  v = ((v >> 8) & 0x00FF00FF00FF00FFULL) | ((v & 0x00FF00FF00FF00FFULL) << 8);
327  v = ((v >> 16) & 0x0000FFFF0000FFFFULL) | ((v & 0x0000FFFF0000FFFFULL) << 16);
328  v = (v >> 32) | (v << 32);
329  return v;
330 }

501 static inline int m4ri_lesser_LSB(word a, word b) {
502  uint64_t const ia = __M4RI_CONVERT_TO_UINT64_T(a);
503  uint64_t const ib = __M4RI_CONVERT_TO_UINT64_T(b);
504  /*
505  * If a is zero then we should always return false, otherwise
506  * if b is zero we should return true iff a has at least one bit set.
507  */
508  return !(ib ? ((ia - 1) ^ ia) & ib : !ia);
509 }

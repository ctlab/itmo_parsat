109  bool checkCap(int new_size) const {
110  return new_size > cap;

113  int32_t index(const K& k) const {
114  return hash(k) % cap;
115  }
116  void _insert(const K& k, const D& d) {
117  vec<Pair>& ps = table[index(k)];
118  ps.push();
119  ps.last().key = k;
120  ps.last().data = d;
121  }

123  void rehash() {
124  const vec<Pair>* old = table;
125 
126  int old_cap = cap;
127  int newsize = primes[0];
128  for (int i = 1; newsize <= cap && i < nprimes; i++)
129  newsize = primes[i];
130 
131  table = new vec<Pair>[newsize];
132  cap = newsize;
133 
134  for (int i = 0; i < old_cap; i++) {
135  for (int j = 0; j < old[i].size(); j++) {
136  _insert(old[i][j].key, old[i][j].data);
137  }
138  }
139 
140  delete[] old;
141 
142  // printf(" --- rehashing, old-cap=%d, new-cap=%d\n", cap, newsize);
143  }

184  if (size == 0)

187  for (int i = 0; i < ps.size(); i++)

195  bool has(const K& k) const {
196  if (size == 0)
197  return false;
198  const vec<Pair>& ps = table[index(k)];
199  for (int i = 0; i < ps.size(); i++)
200  if (equals(ps[i].key, k))
201  return true;
202  return false;

205  // PRECONDITION: the key must exist in the map.
206  void remove(const K& k) {
207  assert(table != NULL);
208  vec<Pair>& ps = table[index(k)];
209  int j = 0;
210  for (; j < ps.size() && !equals(ps[j].key, k); j++)
211  ;
212  assert(j < ps.size());
213  ps[j] = ps.last();
214  ps.pop();
215  size--;
216  }

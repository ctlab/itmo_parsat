30 template <class T>
31 class Queue {
32  vec<T> buf;
33  int first;
34  int end;
35 
36  public:
37  typedef T Key;
38 
39  Queue() : buf(1), first(0), end(0) {}
40 
41  void clear(bool dealloc = false) {
42  buf.clear(dealloc);
43  buf.growTo(1);
44  first = end = 0;
45  }
46  int size() const {
47  return (end >= first) ? end - first : end - first + buf.size();
48  }
49 
50  const T& operator[](int index) const {
51  assert(index >= 0);
52  assert(index < size());
53  return buf[(first + index) % buf.size()];
54  }
55  T& operator[](int index) {
56  assert(index >= 0);
57  assert(index < size());
58  return buf[(first + index) % buf.size()];
59  }
60 
61  T peek() const {
62  assert(first != end);
63  return buf[first];
64  }
65  void pop() {
66  assert(first != end);
67  first++;
68  if (first == buf.size())
69  first = 0;
70  }
71  void copyTo(Queue<T>& copy) const {
72  copy.first = first;
73  copy.end = end;
74  buf.memCopyTo(copy.buf);
75  }
76  void insert(T elem) { // INVARIANT: buf[end] is always unused
77  buf[end++] = elem;
78  if (end == buf.size())
79  end = 0;
80  if (first == end) { // Resize:
81  vec<T> tmp((buf.size() * 3 + 1) >> 1);
82  //**/printf("queue alloc: %d elems (%.1f MB)\n", tmp.size(), tmp.size() * sizeof(T) /
83  //1000000.0);
84  int i = 0;
85  for (int j = first; j < buf.size(); j++)
86  tmp[i++] = buf[j];
87  for (int j = 0; j < end; j++)
88  tmp[i++] = buf[j];
89  first = 0;
90  end = buf.size();
91  tmp.moveTo(buf);
92  }
93  }
94 };

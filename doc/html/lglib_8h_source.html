41 // 'Forking' copies only irredundant clauses and also uses internal variable
42 // indices of the parent as external variable indices. Thus 'parent' and
43 // the forked off 'child' do neither exactly work the same way, nor do they
44 // use from the API point of view the same set of variables. They are
45 // satisfiability equivalent. If the child becomes unsatisfiable the parent
46 // can be assumed to be unsatisfiable too and thus 'lgljoin' would just add
47 // the empty clause to the parent. If the child produces a satisfying
48 // assignment, this assignment is turned into an assignment of the parent by
49 // 'lgljoin'. Parents can be forked multiple times. A child has exactly
50 // one parent. Parents and children can be released independently. Between
51 // 'lglfork' and 'lgljoin' no other operations but further 'lglfork' are
52 // allowed. The effect ot multiple 'lgljoin' with the same parent is
53 // unclear at this point. The same memory manager is use for the child and
54 // the parent. Options, prefix, output file and the callbacks for 'getime'
55 // and 'onabort' are copied too (if set).
56 
57 LGL* lglfork(LGL* parent);
58 int lgljoin(LGL* parent, LGL* child); // does not release 'child'

60 // Both 'Cloning' and 'Forking' can be used to implement 'Push & Pop', but
61 // the asymmetric forking is more similar to the classical way of
62 // implementing it, needs less resources since for instance eliminated
63 // variables do not occupy any memory in the children, but requires lifting
64 // back satisfying assignments explicitly (through the whole parent chain).
65 // If these full satisfying assignments are really needed actually cloning
66 // could be more space efficient too. Assume you want to split the solver
67 // into two instances, one with a literal set to false, the other one to
68 // true. Then cloning allows you to produce two independent branches, while
69 // for forking you need three, since the root / top solver still has to be
70 // kept for lifting back a potential assignment.

192 // Multiple-Objective SAT tries to solve multiple objectives at the same
193 // time. If a target can be satisfied or falsified the user is notified via
194 // a callback function of type 'lglnotify'. The argument is an abstract
195 // state (of the user), the target literal and as last argument the value
196 // (-1 = unsatisfiable, 1 = satisfiable). If the notification callback
197 // returns 1 the 'lglmosat' procedure continues, otherwise it stops. It
198 // returns a non zero value iff only all targets have been either proved to
199 // be satisfiable or unsatisfiable. Even if 'val' given to the notification
200 // function is '1', it can not assume that the current assignment is
201 // complete, e.g. satisfies the formula.
202 //

208 // Incremental interface provides reference counting for indices, i.e.
209 // unfrozen indices become invalid after next 'lglsat' (or 'lglsimp').
210 // This is actually a reference counter for variable indices still in use
211 // after the next 'lglsat' or 'lglsimp' call. It is actually variable based
212 // and only applies to literals in new clauses or used as assumptions,
213 // e.g. in calls to 'lgladd' and 'lglassume'.
214 //
215 // The following example is actually compilable and used for explaining
216 // all the details of this rather complicated incremental API contract:

220 #include "lglib.h"

222 #ifdef NDEBUG

226 #include <assert.h>

228 int main () {
229  LGL * lgl = lglinit ();
230  int res;
231  lgladd (lgl, -14); lgladd (lgl, 2); lgladd (lgl, 0); // binary clause
232  lgladd (lgl, 14); lgladd (lgl, -1); lgladd (lgl, 0); // binary clause
233  lglassume (lgl, 1); // assume '1'
234  lglfreeze (lgl, 1); // will use '1' below
235  lglfreeze (lgl, 14); // will use '14 too
236  assert (lglfrozen (lgl, 1));
237  assert (lglfrozen (lgl, 14));
238  res = lglsat (lgl);
239  assert (res == 10);
240  (void) lglderef (lgl, 1); // fine
241  (void) lglderef (lgl, 2); // fine
242  (void) lglderef (lgl, 3); // fine !
243  (void) lglderef (lgl, 14); // fine
244  assert (!lglusable (lgl, 2));
245  // lgladd (lgl, 2); // ILLEGAL
246  // lglassume (lgl, 2); // ILLEGAL
247  assert (lglusable (lgl, 15)); // '15' not used yet!
248  lgladd (lgl, -14); lgladd (lgl, 1); lgladd (lgl, 0); // binary clause
249  lgladd (lgl, 15); lgladd (lgl, 0); // fine!
250  lglmelt (lgl, 14); // '1' discarded
251  res = lglsat (lgl);
252  assert (res == 10);
253  assert (lglfrozen (lgl, 1));
254  (void) lglderef (lgl, 1); // fine
255  (void) lglderef (lgl, 2); // fine
256  (void) lglderef (lgl, 3); // fine
257  (void) lglderef (lgl, 14); // fine
258  (void) lglderef (lgl, 15); // fine
259  assert (!lglusable (lgl, 2));
260  assert (!lglusable (lgl, 14));
261  // lglassume (lgl, 2); // ILLEGAL
262  // lglassume (lgl, 14); // ILLEGAL
263  lgladd (lgl, 1); // still frozen
264  lglmelt (lgl, 1);
265  lgladd (lgl, 0);
266  res = lglsat (lgl);
267  // none frozen
268  assert (!lglusable (lgl, 1));
269  // lgladd(lgl, 1); // ILLEGAL
270  lglsetopt (lgl, "plain", 1); // disable BCE ...
271  lgladd (lgl, 8); lgladd (lgl, -9); lgladd (lgl, 0);
272  res = lglsat (lgl);
273  assert (res == 10);
274  assert (!lglusable (lgl, 8));
275  assert (!lglusable (lgl, -9));
276  assert (lglreusable (lgl, 8));
277  assert (lglreusable (lgl, -9));
278  lglreuse (lgl, 8);
279  lgladd (lgl, -8); lgladd (lgl, 0);
280  lglsetopt (lgl, "plain", 0); // enable BCE ...
281  res = lglsat (lgl);
282  assert (res);
283  return res;
284 }
